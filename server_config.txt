# systemd: /etc/systemd/system/flask_app.service
[Unit]
Description=Gunicorn for Flask app
After=network.target

[Service]
User=ubuntu
Group=www-data
WorkingDirectory=/var/www/html
Environment="PATH=/var/www/html/venv/bin"
Environment="SECRET_KEY=<SECRET_KEY>"
Environment="ADMIN_USERNAME=admin"
Environment="ADMIN_PASSWORD_HASH=<ADMIN_PASSWORD_HASH>"
Environment="MAIL_USERNAME=<MAIL_USERNAME>"
Environment="MAIL_PASSWORD=<MAIL_PASSWORD>"
Environment="MAIL_DEFAULT_SENDER=<MAIL_DEFAULT_SENDER>"
Environment="S3_BUCKET=ismailtrache-uploads"
Environment="S3_REGION=us-east-1"
Environment="S3_PUBLIC_BASE=https://d3niujx36r14qz.cloudfront.net"
Environment="S3_PREFIX=uploads"
Environment="S3_BACKUP_PREFIX=backups"
Environment="FORCE_HTTPS=1"
ExecStart=/var/www/html/venv/bin/gunicorn -w 2 -b 127.0.0.1:8000 app:app

[Install]
WantedBy=multi-user.target

# nginx: /etc/nginx/sites-available/app
server {
    listen 80;
    server_name ismailtrache.me www.ismailtrache.me;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl http2;
    server_name ismailtrache.me www.ismailtrache.me;

    ssl_certificate /etc/letsencrypt/live/ismailtrache.me/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/ismailtrache.me/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains; preload" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-Frame-Options "DENY" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
    add_header Permissions-Policy "geolocation=()" always;

    location / {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
